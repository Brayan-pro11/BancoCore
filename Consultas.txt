1. 3 consultas usando sub consultas.

-- Préstamos que pertenecen a clientes existentes

SELECT 
    ID_Prestamo,
    ID_Cliente_Solicitante,
    Monto_Solicitado
FROM Prestamo
WHERE ID_Cliente_Solicitante IN (
    SELECT ID_Identificacion
    FROM Cliente_Persona_Natural
);


-- Cuentas que pertenecen a clientes que tienen préstamos

SELECT 
    Numero_Cuenta,
    ID_Titular,
    Saldo_Actual
FROM Cuenta_Bancaria
WHERE ID_Titular IN (
    SELECT ID_Cliente_Solicitante
    FROM Prestamo
);


-- Mostra las personas que sí tienen préstamo registrado

SELECT nombre_completo, id_identificacion
FROM cliente_persona_natural
WHERE id_identificacion IN (
    SELECT id_cliente_solicitante
    FROM prestamo
);

2. 3 consultas usando filtros de igualacion

-- Mostrar todos los préstamos aprobados (si existe al menos uno)

SELECT id_prestamo, id_cliente_solicitante, monto_solicitado
FROM prestamo
WHERE estado_prestamo = 'Aprobado';


-- Mostrar préstamos con monto exactamente igual al monto más alto

SELECT id_prestamo, monto_solicitado
FROM prestamo
WHERE monto_solicitado = (
    SELECT MAX(monto_solicitado)
    FROM prestamo
);

-- Mostrar clientes que tengan exactamente el mismo ID que el primer préstamo registrado

SELECT nombre_completo, id_identificacion
FROM cliente_persona_natural
WHERE id_identificacion = (
    SELECT id_cliente_solicitante
    FROM prestamo
    LIMIT 1
);

3. 3 contultas usando join

-- Mostrar clientes con sus préstamos

SELECT 
    c.nombre_completo,
    c.id_identificacion,
    p.id_prestamo,
    p.monto_solicitado,
    p.estado_prestamo
FROM cliente_persona_natural c
JOIN prestamo p 
    ON c.id_identificacion = p.id_cliente_solicitante;

-- Mostrar préstamos junto con el correo del cliente

SELECT 
    p.id_prestamo,
    p.monto_solicitado,
    p.estado_prestamo,
    c.correo_electronico
FROM prestamo p
JOIN cliente_persona_natural c
    ON p.id_cliente_solicitante = c.id_identificacion;

-- Mostrar el total de préstamos por cliente

SELECT 
    c.nombre_completo,
    COUNT(p.id_prestamo) AS total_prestamos
FROM cliente_persona_natural c
JOIN prestamo p
    ON c.id_identificacion = p.id_cliente_solicitante
GROUP BY c.nombre_completo;

4. 3 consultas usando left join

-- Mostrar clientes que NO tienen préstamo

SELECT 
    c.nombre_completo,
    c.id_identificacion
FROM cliente_persona_natural c
LEFT JOIN prestamo p
    ON c.id_identificacion = p.id_cliente_solicitante
WHERE p.id_prestamo IS NULL;

-- Total de dinero solicitado por cada cliente (aunque no tenga préstamos)
 
SELECT 
    c.nombre_completo,
    c.id_identificacion,
    IFNULL(SUM(p.monto_solicitado), 0) AS total_solicitado
FROM cliente_persona_natural c
LEFT JOIN prestamo p
    ON c.id_identificacion = p.id_cliente_solicitante
GROUP BY c.id_identificacion, c.nombre_completo;

-- Transferencias con usuario creador y aprobador

SELECT 
    t.ID_Transferencia,
    u1.Nombre_Completo AS Creador,
    u2.Nombre_Completo AS Aprobador,
    t.Estado_Transferencia
FROM Transferencia t
JOIN Usuarios u1
    ON t.ID_Usuario_Creador = u1.ID_Usuario
JOIN Usuarios u2
    ON t.ID_Usuario_Aprobador = u2.ID_Usuario;

5. 3 consultas usando rigth join

-- Usuarios y todas las transferencias (aunque no las hayan creado)

SELECT 
    u.Nombre_Completo,
    t.ID_Transferencia,
    t.Estado_Transferencia
FROM Transferencia t
RIGHT JOIN Usuarios u
    ON t.ID_Usuario_Creador = u.ID_Usuario;


-- Transferencias y fecha de creación por usuario

SELECT 
    u.Nombre_Completo,
    t.ID_Transferencia,
    t.Fecha_Creacion
FROM Transferencia t
RIGHT JOIN Usuarios u
    ON t.ID_Usuario_Creador = u.ID_Usuario
ORDER BY u.Nombre_Completo;

-- Mostrar todas las cuentas bancarias y préstamos que fueron desembolsados a ellas

Mostrando todas las cuentas, aunque no tengan préstamos asociados:

SELECT 
    c.Numero_Cuenta,
    c.ID_Titular,
    pr.ID_Prestamo,
    pr.Monto_Aprobado,
    pr.Fecha_Desembolso
FROM Prestamo pr
RIGHT JOIN Cuenta_Bancaria c
    ON pr.Cuenta_Destino_Desembolso = c.Numero_Cuenta
ORDER BY c.Numero_Cuenta;
