1) CASO 1: Carga masiva de transferencias, las transferencias se realizan uno a uno hasta que no hayan fondos, alli deben detenerse.
2) Caso 2: Varias operaciones pendientes de aprobacion, con las aplicaciones masivas en los escenarios 1 a 1 donde la persona esta aprobando las transferencias, cuando una transferencia se realice y deje sin fondo a las demas, debe detenerse.
-------------------------------------------------------------------------------------------------------------------------------------------

-- TRIGGER CASO 1 

DELIMITER $$

CREATE TRIGGER trg_validar_fondos_insert
BEFORE INSERT ON transferencia
FOR EACH ROW
BEGIN
    DECLARE saldo_actual DECIMAL(18,2);

    -- Obtener saldo de la cuenta origen
    SELECT Saldo_Actual
    INTO saldo_actual
    FROM cuenta_bancaria
    WHERE Numero_Cuenta = NEW.Cuenta_Origen;

    -- Validar fondos
    IF saldo_actual < NEW.Monto THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Fondos insuficientes para realizar la transferencia';
    ELSE
        -- Descontar saldo
        UPDATE cuenta_bancaria
        SET Saldo_Actual = Saldo_Actual - NEW.Monto
        WHERE Numero_Cuenta = NEW.Cuenta_Origen;
    END IF;

END$$

DELIMITER ;

-- TRIGGER CASO 2 

DELIMITER $$

CREATE TRIGGER trg_validar_fondos_update
BEFORE UPDATE ON transferencia
FOR EACH ROW
BEGIN
    DECLARE saldo_actual DECIMAL(18,2);

    -- Solo si se está aprobando / ejecutando
    IF NEW.Estado_Transferencia = 'Ejecutada' 
       AND OLD.Estado_Transferencia <> 'Ejecutada' THEN

        -- Obtener saldo actual
        SELECT Saldo_Actual
        INTO saldo_actual
        FROM cuenta_bancaria
        WHERE Numero_Cuenta = NEW.Cuenta_Origen;

        -- Validar fondos
        IF saldo_actual < NEW.Monto THEN
            SIGNAL SQLSTATE '45000'
            SET MESSAGE_TEXT = 'Fondos insuficientes para aprobar la transferencia';
        ELSE
            -- Descontar saldo
            UPDATE cuenta_bancaria
            SET Saldo_Actual = Saldo_Actual - NEW.Monto
            WHERE Numero_Cuenta = NEW.Cuenta_Origen;
        END IF;

    END IF;

END$$

DELIMITER ;
